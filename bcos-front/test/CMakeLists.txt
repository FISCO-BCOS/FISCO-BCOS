file(GLOB_RECURSE SOURCES "*.cpp")

# cmake settings
set(TEST_BINARY_NAME test-bcos-front)

add_executable(${TEST_BINARY_NAME} ${SOURCES})
target_include_directories(${TEST_BINARY_NAME} PRIVATE .)
find_package(Boost REQUIRED unit_test_framework)
target_link_libraries(${TEST_BINARY_NAME} PUBLIC ${FRONT_TARGET} ${TARS_PROTOCOL_TARGET} Boost::unit_test_framework)
set_source_files_properties("unittests/FrontServiceTest.cpp" PROPERTIES SKIP_UNITY_BUILD_INCLUSION ON)
set_target_properties(${TEST_BINARY_NAME} PROPERTIES UNITY_BUILD "ON")
config_test_cases("" "${SOURCES}" "${TEST_BINARY_NAME}" "" "")
